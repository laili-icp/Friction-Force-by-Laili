    async function captureAndAnalyze() {
        const context = canvas.getContext('2d');
        canvas.width = video.videoWidth;
        canvas.height = video.videoHeight;
        context.drawImage(video, 0, 0, canvas.width, canvas.height);

        // Convert gambar ke Base64 untuk dikirim ke AI
        const imageData = canvas.toDataURL('image/jpeg').split(',')[1];

        document.getElementById('loading-spinner').style.display = "block";
        document.getElementById('result-display').style.display = "none";

        // PERINTAH UNTUK AI (Prompt)
        const prompt = "Analisis gambar ini. Sebutkan nama bahan benda ini, prediksi tingkat kekasaran permukaannya (Skala 1-10), dan berikan penjelasan singkat kenapa permukaannya kasar/halus berdasarkan teksturnya.";

        try {
            const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${API_KEY}`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    contents: [{
                        parts: [
                            { text: prompt },
                            { inline_data: { mime_type: "image/jpeg", data: imageData } }
                        ]
                    }]
                })
            });

            const data = await response.json();
            const aiText = data.candidates[0].content.parts[0].text;

            // Tampilkan hasil
            displayResult(aiText);
        } catch (error) {
            alert("Gagal koneksi ke internet/AI. Cek API Key Anda.");
        } finally {
            document.getElementById('loading-spinner').style.display = "none";
        }
    }
